conversation
    conversationId (unique)
    conversation_alias (optional, string)

conversation_party
    conversationId (many)
    userId (many)
    notifications (boolean - flag notification if new message appears, default TRUE)

conversation_entry
    conversation_entry_id (unique)
    conversation_id (many)
    user_id (many)
    entry_date (for sorting & display)
    entry (string, encrypted)
    entry_ref (number)

user
    userId (unique)
    username (unique)
    password (string, encrypted)
    status (int)
        0 = offline
        1 = online
        2 = away
